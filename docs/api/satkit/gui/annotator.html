 65class Annotator():
 66    """
 67    Annotator controls QtAnnotatorWindow and the data.
 68
 69    In the Model-view-presenter pattern this class is the Presenter,
 70    QtAnnotatorWindow is the View and a Recordings array is the Model.
 71    """
 72
 73    default_categories = ['Stable', 'Hesitation', 'Chaos', 'No data',
 74                          'Not set']
 75
 76    default_tongue_positions = ['High', 'Low', 'Other / Not visible']
 77
 78    def __init__(self, recordings, args, xlim=(-0.1, 1.0),
 79                 categories=None, pickle_filename=None):
 80
 81        self.view_window = QtAnnotatorWindow(self)
 82
 83        self.index = 0
 84        self.max_index = len(recordings)
 85
 86        self.recordings = recordings
 87        self.commandlineargs = args
 88        self.display_tongue = args.displayTongue
 89
 90        self.xlim = xlim
 91
 92        if categories is None:
 93            self.categories = Annotator.default_categories
 94        else:
 95            self.categories = categories
 96        self.tongue_positions = Annotator.default_tongue_positions
 97        self._add_annotations()
 98
 99        self.pickle_filename = pickle_filename
100
101        self.has_been_saved = False
102
103    @property
104    def current(self):
105        """Current recording."""
106        return self.recordings[self.index]
107
108    @property
109    def default_annotations(self):
110        """Default annotations with default values."""
111        return {
112            'pdCategory': self.categories[-1],
113            'tonguePosition': self.tongue_positions[-1],
114            'pdOnset': -1.0,
115            'pdOnsetIndex': -1,
116        }
117
118    def _add_annotations(self):
119        """Add annotation dicts to this annotator's recordings."""
120        for recording in self.recordings:
121            if recording.annotations:
122                recording.annotations = dict(
123                    list(self.default_annotations.items()) +
124                    list(recording.annotations.items()))
125            else:
126                recording.annotations = deepcopy(self.default_annotations)
127
128    def next_cb(self):
129        """
130        Callback function for the Next button.
131        Increases cursor index, updates the view.
132        """
133        if self.index < self.max_index-1:
134            # TODO: wrap in a data modalities accessor
135            self.current.modalities['RawUltrasound'].data = None
136            self.index += 1
137            self.view_window.update_figures()
138            self.view_window.update_ui()
139
140    def prev_cb(self):
141        """
142        Callback function for the Previous button.
143        Decreases cursor index, updates the view.
144        """
145        if self.index > 0:
146            # TODO: wrap in a data modalities accessor
147            self.current.modalities['RawUltrasound'].data = None
148            self.index -= 1
149            self.view_window.update_figures()
150            self.view_window.update_ui()
151
152    def go_cb(self):
153        """
154        Go to a recording.
155        """
156        self.current.modalities['RawUltrasound'].data = None
157        self.index = int(self.view_window.goLineEdit.text())-1
158        self.view_window.update_figures()
159        self.view_window.update_ui()
160
161    def quit_cb(self):
162        """
163        Quit the app.
164        """
165        # TODO: make sure the data has been saved before quitting.
166        if not self.has_been_saved:
167            self.save_cb()
168        QCoreApplication.quit()
169
170    def save_cb(self):
171        """
172        Save the recordings.
173        """
174        if not self.pickle_filename:
175            (self.pickle_filename, _) = QFileDialog.getSaveFileName(
176                self, 'Save file', directory='.', filter="Pickle files (*.pickle)")
177        if self.pickle_filename:
178            # satkit_io.save2pickle(
179            #     self.recordings,
180            #     self.pickle_filename)
181            _qt_annotator_logger.info(
182                "Pickling is disabled. Did NOT write file {file}.", file=self.pickle_filename)
183        self.has_been_saved = True
184
185    def export_cb(self):
186        """
187        Export annotations and some other meta data to a utf-8 .csv file.
188        """
189        (filename, _) = QFileDialog.getSaveFileName(
190            self, 'Save file', directory='.', filter="CSV files (*.csv)")
191
192        if not filename:
193            return
194
195        vowels = ['a', 'A', 'e', 'E', 'i', 'I',
196                  'o', 'O', 'u', '@', "@`", 'OI', 'V']
197        fieldnames = ['basename', 'date_and_time', 'prompt', 'C1', 'C1_dur',
198                      'word_dur', 'first_sound',
199                      'first_sound_type', 'first_sound_dur', 'AAI']
200        fieldnames.extend(self.default_annotations.keys())
201        csv.register_dialect('tabseparated', delimiter='\t',
202                             quoting=csv.QUOTE_NONE)
203
204        with closing(open(filename, 'w', encoding='utf-8')) as csvfile:
205            writer = csv.DictWriter(csvfile, fieldnames, extrasaction='ignore',
206                                    dialect='tabseparated')
207
208            writer.writeheader()
209            for recording in self.recordings:
210                annotations = recording.annotations.copy()
211                annotations['basename'] = recording.meta['basename']
212                annotations['date_and_time'] = recording.meta['date_and_time']
213                annotations['prompt'] = recording.meta['prompt']
214                annotations['word'] = recording.meta['prompt'].split()[1]
215
216                word_dur = -1.0
217                acoustic_onset = -1.0
218                if 'word' in recording.textgrid:
219                    for interval in recording.textgrid['word']:
220                        # change this to access the phonemeDict and check for included words,
221                        # then search for phonemes based on the same
222                        if interval.text == "":
223                            continue
224
225                        # Before 1.0: check if there is a duration to use here. and maybe make this
226                        # more intelligent by selecting purposefully the last non-empty first and
227                        # taking the duration?
228                        word_dur = interval.dur
229                        stimulus_onset = recording['MonoAudio'].meta['stimulus_onset']
230                        acoustic_onset = interval.xmin - stimulus_onset
231                        break
232                    annotations['word_dur'] = word_dur
233                else:
234                    annotations['word_dur'] = -1.0
235
236                if acoustic_onset < 0 or annotations['pdOnset'] < 0:
237                    aai = -1.0
238                else:
239                    aai = acoustic_onset - annotations['pdOnset']
240                annotations['AAI'] = aai
241
242                first_sound_dur = -1.0
243                first_sound = ""
244                if 'segment' in recording.textgrid:
245                    for interval in recording.textgrid['segment']:
246                        if interval.text and interval.text != 'beep':
247                            first_sound_dur = interval.dur
248                            first_sound = interval.text
249                            break
250                annotations['first_sound_dur'] = first_sound_dur
251                annotations['first_sound'] = first_sound
252                if first_sound in vowels:
253                    annotations['first_sound_type'] = 'V'
254                else:
255                    annotations['first_sound_type'] = 'C'
256
257                annotations['C1'] = recording.meta['prompt'][0]
258                writer.writerow(annotations)
259            _qt_annotator_logger.info(
260                'Wrote onset data in file {file}.', file=filename)
261
262    def pd_category_cb(self):
263        """
264        Callback funtion for the RadioButton for catogorising
265        the PD curve.
266        """
267        radio_button = self.view_window.sender()
268        if radio_button.isChecked():
269            self.current.annotations['pdCategory'] = radio_button.text()
270            self.has_been_saved = False
271
272    def tongue_position_cb(self):
273        """
274        Callback funtion for the RadioButton for catogorising
275        the PD curve.
276        """
277        radio_button = self.view_window.sender()
278        if radio_button.isChecked():
279            self.current.annotations['tonguePosition'] = radio_button.text()
280            self.has_been_saved = False
281
282    def onpick_cb(self, event):
283        """
284        Callback for handling time selection on events.
285        """
286        subplot = 0
287        for i, axis in enumerate([self.view_window.ax1]):
288            # For infomation, print which axes the click was in
289            if axis == event.inaxes:
290                subplot = i+1
291                break
292
293        if subplot == 1:
294            self.current.annotations['pdOnset'] = event.pickx
295
296            audio = self.current.modalities['MonoAudio']
297            stimulus_onset = audio.meta['stimulus_onset']
298
299            pd = self.current.modalities['PD on RawUltrasound']
300            ultra_time = pd.timevector - stimulus_onset
301            self.current.annotations['pdOnsetIndex'] = np.nonzero(
302                ultra_time >= event.pickx)[0][0]
303            self.has_been_saved = False
304        self.view_window.update_figures()
