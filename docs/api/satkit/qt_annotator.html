255    def _add_annotations(self):
256        """Add the annotations."""
257        for recording in self.recordings:
258            if recording.annotations:

260                    list(self.default_annotations.items()) +
261                    list(recording.annotations.items()))

484    def draw_ultra_frame(self):
485        """
486        Display an already interpolated ultrasound frame.
487        """
488        if (self.current.excluded or
489                self.current.annotations['selection_index'] == -1):
490            self.ultra_axes.clear()
491        elif self.current.annotations['selection_index']:
492            self.ultra_axes.clear()
493            index = self.current.annotations['selection_index']
494
495            ultrasound = self.current.modalities['RawUltrasound']
496            image = ultrasound.interpolated_image(index)
497
498            self.ultra_axes.imshow(
499                image, interpolation='nearest', cmap='gray',
500                extent=(-image.shape[1]/2-.5, image.shape[1]/2+.5,
501                        -.5, image.shape[0]+.5))
502
503            if 'Splines' in self.current.modalities:
504                splines = self.current.modalities['Splines']
505                index = self.current.annotations['selection_index']
506                ultra = self.current.modalities['RawUltrasound']
507                timestamp = ultra.timevector[index]
508
509                spline_index = np.argmin(
510                    np.abs(splines.timevector - timestamp))
511
512                # TODO: move this to reading splines/end of loading and make
513                # the system warn the user when there is a creeping
514                # discrepancy. also make it a integration test where
515                # spline_test_token1 gets run and triggers this
516                # ic(splines.timevector)
517                # ic(ultra.timevector[:len(splines.timevector)])
518                # time_diff = splines.timevector - \
519                #     ultra.timevector[:len(splines.timevector)]
520                # ic(np.diff(time_diff, n=1))
521                # ic(np.max(np.abs(np.diff(time_diff, n=1))))
522
523                epsilon = max((config_dict['epsilon'], splines.time_precision))
524                min_difference = abs(
525                    splines.timevector[spline_index] - timestamp)
526                # maybe this instead when loading data
527                # str(number)[::-1].find('.') -> precision
528
529                # ic(epsilon, splines.timevector[spline_index] - timestamp)
530                # ic(splines.timevector[spline_index], timestamp)
531                if min_difference > epsilon:
532                    _logger.info("Splines out of synch in %s.",
533                                 self.current.basename)
534                    _logger.info("Minimal difference: %f, epsilon: %f",
535                                 min_difference, epsilon)
536
537                spline_config = self.recording_session.config.spline_config
538                if spline_config.data_config:
539                    limits = spline_config.data_config.ignore_points
540                    plot_spline(self.ultra_axes,
541                                splines.cartesian_spline(spline_index),
542                                limits=limits)
543                else:
544                    plot_spline(self.ultra_axes,
545                                splines.cartesian_spline(spline_index))
546            else:
547                _logger.info("No splines")
548        self.ultra_canvas.draw()

550    def draw_raw_ultra_frame(self):
551        """
552        Interpolate and display a raw ultrasound frame.
553        """
554        if self.current.annotations['selection_index']:
555            ind = self.current.annotations['selection_index']
556            array = self.current.modalities['RawUltrasound'].data[ind, :, :]
557        else:
558            array = self.current.modalities['RawUltrasound'].data[1, :, :]
559        array = np.transpose(array)
560        array = np.flip(array, 0).copy()
561        array = array.astype(np.int8)
562        self.ultra_axes.imshow(array, interpolation='nearest', cmap='gray')
563        self.ultra_canvas.draw()
