satkit.configuration.configuration_models

These are the code friendly wrappers for the configuration structures.

configuration_parsers contains the actual parsing of strictyaml into comment-retaining dictionary-like structures. Here those structures get parsed into pydantic models that know what their fields actually are.

This two level structure is maintained so that in some version after SATKIT 1.0 we can implement configuration round tripping with preserved comments.

 71class SearchPattern(UpdatableBaseModel):
 72    pattern: str
 73    is_regexp: Optional[bool] = False
 74
 75    @staticmethod
 76    def build(value: Union[dict, str]) -> 'SearchPattern':
 77
 78        if isinstance(value, str):
 79            return SearchPattern(pattern=value)
 80
 81        if isinstance(value, dict):
 82            return SearchPattern(**value)
 83
 84        raise ValueError("Unrecognised input value type: "
 85                         + type(value).__name__ + ".")

 88class TimeseriesNormalisation(UpdatableBaseModel):
 89    """
 90
 91    """
 92    peak: Optional[bool] = False
 93    bottom: Optional[bool] = False
 94
 95    # TODO: this class needs a special dumper to save things correctly.
 96
 97    @staticmethod
 98    def build(
 99            value: str) -> 'TimeseriesNormalisation':
100        """
101        Construct a TimeseriesNormalisation object from a string value.
102
103        The value usually comes from a config file.
104
105        Parameters
106        ----------
107        value : str
108            'none': no normalisation
109            'peak': divide all data points y-values by the largest y-value
110            'bottom': deduct the lowest y-value from all data points y-values
111            'both': do first bottom normalisation and then peak normalisation.
112
113        Returns
114        -------
115        TimeseriesNormalisation
116            The new TimeseriesNormalisation with fields set as expected.
117        """
118        print(value)
119        match value:
120            case 'none':
121                print(value)
122                return TimeseriesNormalisation()
123            case 'peak':
124                print(value)
125                return TimeseriesNormalisation(peak=True)
126            case 'bottom':
127                print(value)
128                return TimeseriesNormalisation(bottom=True)
129            case 'both':
130                print(value)
131                return TimeseriesNormalisation(peak=True, bottom=True)
132            case _:
133                raise ValueError("Unrecognised value: " + value + ".")
