 89@dataclass
 90class PointAnnotations():
 91    """
 92    Time point annotations for a Modality.
 93
 94    For each modality there should be only one of these for each kind of
 95    annotation type. 
 96
 97    annotation_type : AnnotationType
 98        unique identifier for the annotation type
 99    indeces : np.ndarray
100        indeces of the annotation points. `modality_data.data[indeces[i]]` and
101        `modality_data.timevector[indeces[i]]` correspond to the annotation at
102        `i`.
103    times : np.ndarray 
104        timestamps of the annotation points
105    generating_parameters : dict 
106        the function call arguments and other parameters used in generating
107        these annotations.
108    properties : dict
109        a dictionary containing arrays of each of the annotation properties
110        expected for this annotation type.
111    """
112    annotation_type: AnnotationType
113    indeces: np.ndarray
114    times: np.ndarray
115    generating_parameters: UpdatableBaseModel
116    properties: dict

121        This method has not been implemented yet.

144    def apply_lower_time_limit(self, time_min: float) -> None:
145        """
146        Apply a lower time limit to the annotations.
147
148        This removes the annotation points before the given time limit.
149
150        Parameters
151        ----------
152        time_min : float
153            The time limit.
154        """
155        selected = np.nonzero(self.times >= time_min)
156        self.indeces = self.indeces[selected]
157        self.times = self.times[selected]
158        limit = selected[0]
159
160        for key in self.properties:
161            if is_sequence_form(self.properties[key]):
162                ic(key, self.properties[key])
163                self.properties[key] = self.properties[key][limit:]
164            elif isinstance(self.properties[key], np.ndarray):
165                self.properties[key] = self.properties[key][selected]
166
167    def apply_upper_time_limit(self, time_max: float) -> None:
168        """
169        Apply an upper time limit to the annotations.
170
171        This removes the annotation points after the given time limit.
172
173        Parameters
174        ----------
175        time_max : float
176            The time limit.
177        """
178        selected = np.nonzero(self.times <= time_max)
179        self.indeces = self.indeces[selected]
180        self.times = self.times[selected]
181        limit = selected[-1]
182
183        for key in self.properties:
184            if is_sequence_form(self.properties[key]):
185                ic(key, self.properties[key])
186                self.properties[key] = self.properties[key][:limit]
187            elif isinstance(self.properties[key], np.ndarray):
188                self.properties[key] = self.properties[key][selected]

234class Recording(UserDict):
235    """
236    A Recording is a dictionary of 0-n synchronised Modalities.
237
238    Adding modalities can be done by `recording[name] = modality`, but
239    `recording.add_modality` is preferred as a safer way which checks for
240    overwriting. The reason Recording is a dictionary is to make it possible to
241    iterate with the idiom `for modality_name in recording`.
242
243    The recording also contains the non-modality-specific metadata
244    (participant, speech content, etc) as a dictionary, as well as the textgrid
245    for the whole recording.
246
247    In general, inheriting should not be necessary, but if it is, inheriting
248    classes should call self._read_textgrid() after calling super.__init__()
249    (with correct arguments) and doing any updates to self.meta['textgrid']
250    that are necessary.
251    """
252
253    def __init__(self,
254                 meta_data: RecordingMetaData,
255                 excluded: bool = False,
256                 textgrid_path: Union[str, Path] = "") -> None:
257        """
258        Construct a mainly empty recording without modalities.
259
260        Modalities and annotations get added after constructions with their own
261        add_[modality or annotation] functions.
262
263        Parameters
264        ----------
265        meta_data : RecordingMetaData
266            Some of the contents of the meta data are available as properties.
267        excluded : bool, optional
268            _description_, by default False
269        textgrid_path : Union[str, Path], optional
270            _description_, by default ""
271        """
272        super().__init__()
273
274        self.excluded = excluded
275
276        self.meta_data = meta_data
277
278        self.textgrid_path = textgrid_path
279        if not self.textgrid_path:
280            self.textgrid_path = meta_data.path.joinpath(
281                meta_data.basename + ".TextGrid")
282        self.textgrid = self._read_textgrid()
283        self.satgrid = SatGrid(self.textgrid)
284
285        self.annotations = {}
286
287    @property
288    def modalities(self) -> dict[str, 'Modality']:
289        """
290        Dictionary of the modalities of this Recording.
291
292        Returns
293        -------
294        dict[str, Modality]
295            The dictionary of the Modalities.
296        """
297        return self.data
298
299    @property
300    def path(self) -> Path:
301        """Path to this recording."""
302        return self.meta_data.path
303
304    @path.setter
305    def path(self, path: Path) -> None:
306        self.meta_data.path = path
307
308    @property
309    def basename(self) -> str:
310        """Filename of this Recording without extensions."""
311        return self.meta_data.basename
312
313    @basename.setter
314    def basename(self, basename: str) -> None:
315        self.meta_data.basename = basename

317    def _read_textgrid(self) -> textgrids.TextGrid:
318        """
319        Read the textgrid specified in self.meta.
320
321        If file does not exist or reading fails, recovery is attempted 
322        by logging an error and creating an empty textgrid for this 
323        recording.
324        """
325        textgrid = None
326        if self.textgrid_path.is_file():
327            try:
328                textgrid = textgrids.TextGrid(self.textgrid_path)
329                _datastructures_logger.info("Read textgrid in %s.",
330                                            self.textgrid_path)
331            except Exception as exception:
332                _datastructures_logger.critical(
333                    "Could not read textgrid in %s.", self.textgrid_path)
334                _datastructures_logger.critical(
335                    "Failed with: %s.", str(exception))
336                _datastructures_logger.critical("Creating an empty textgrid "
337                                                + "instead.")
338                textgrid = textgrids.TextGrid()
339        else:
340            notice = 'Note: ' + str(self.textgrid_path) + " did not exist."
341            _datastructures_logger.warning(notice)
342            _datastructures_logger.warning("Creating an empty textgrid "
343                                           + "instead.")
344            textgrid = textgrids.TextGrid()
345        return textgrid
346
347    def identifier(self) -> str:
348        """
349        Generate a unique identifier for this Recording from metadata.
350
351        Returns
352        -------
353        str
354            prompt followed by time of recording.
355        """
356        return (f"{self.meta_data.prompt} "
357                f"{str(self.meta_data.time_of_recording)}")
358
359    def exclude(self) -> None:
360        """
361        Set self.excluded to True with a method.
362
363        This method exists to facilitate list comprehensions being used
364        for excluding recordings e.g. 
365        [recording.exclude() for recording in recordings if in some_list].
366        """
367        self.excluded = True
368
369    def write_textgrid(self, filepath: Path = None) -> None:
370        """
371        Save this recording's textgrid to file.
372
373        Keyword argument:
374        filepath -- string specifying the path and name of the 
375            file to be written. If filepath is not specified, this 
376            method will try to overwrite the textgrid specified in 
377            self.meta.
378
379            If filepath is specified, subsequent calls to this 
380            function will write into the new path rather than 
381            the original one.
382        """
383        try:
384            if filepath:
385                self.textgrid_path = filepath
386                self.textgrid.write(filepath)
387            else:
388                self.textgrid.write(self.textgrid_path)
389        except Exception as exception:
390            _datastructures_logger.critical(
391                "Could not write textgrid to %s.", str(self.textgrid_path))
392            _datastructures_logger.critical(
393                "TextGrid save failed with error: %s", str(exception))
394
395    def add_modality(
396            self, modality: 'Modality', replace: bool = False) -> None:
397        """
398        This method adds a new Modality object to the Recording.
399
400        Replacing a modality has to be specified otherwise if a
401        Modality with the same name already exists in this Recording
402        and the replace argument is not True, an Error is raised. 
403
404        Arguments:
405        modality -- object of type Modality to be added to 
406            this Recording.
407
408        Keyword arguments:
409        replace -- a boolean indicating if an existing Modality should
410            be replaced.
411        """
412        name = modality.name
413
414        if name in self.modalities and not replace:
415            ic(modality.metadata)
416            ic(self[modality.name].metadata)
417            raise ModalityError(
418                "A modality named " + name +
419                " already exists and replace flag was False.")
420
421        if replace:
422            self.modalities[name] = modality
423            _datastructures_logger.debug("Replaced modality %s.", name)
424        else:
425            self.modalities[name] = modality
426            _datastructures_logger.debug("Added new modality %s.", name)
427
428    def __str__(self) -> str:
429        """
430        Bare minimum string representation of the Recording.
431
432        Returns
433        -------
434        string
435            'Recording [basename]'
436        """
437        return f"Recording {self.basename}"

440class Modality(abc.ABC, OrderedDict):
441    """
442    Abstract superclass for all data Modality classes.
443
444    Any annotations associated with a Modality instance are accessible directly
445    by `modality[annotation_type]`, because a Modality is also a OrderedDict of
446    its Annotations.
447    """
448
449    @classmethod
450    @abc.abstractmethod
451    def generate_name(cls, params: ModalityMetaData) -> str:
452        pass
453
454    def __init__(self,
455                 recording: Recording,
456                 parsed_data: Optional[ModalityData] = None,
457                 metadata: Optional[ModalityMetaData] = None,
458                 data_path: Optional[Path] = None,
459                 meta_path: Optional[Path] = None,
460                 load_path: Optional[Path] = None,
461                 time_offset: Optional[float] = None,
462                 annotations: Optional[dict[AnnotationType,
463                                            PointAnnotations]] = None
464                 ) -> None:
465        """
466        Modality constructor.
467
468        Positional arguments:
469        recording -- the containing Recording.
470
471        Keyword arguments:
472        data_path -- path of the data file
473        load_path -- path of data when saved by SATKIT - both data and metadata
474        parent -- the Modality this one was derived from. None means this 
475            is an underived data Modality.
476        parsed_data -- ModalityData object containing waveform, sampling rate,
477            and either timevector and/or time_offset. 
478        parsed_data -- a ModalityData object containing parsed data 
479            that's been either read from file, loaded from file 
480            (previously saved by SATKIT), or calculated from another modality.
481            Providing a timevector 
482            overrides any time_offset value given, but in absence of a 
483            timevector the time_offset will be applied on reading the data 
484            from file. 
485        time_offset -- offset of this modality in relation to the Recordings
486            baseline - usually the audio track. This will be ignored if 
487            parsed_data exists and effectively overridden by 
488            parsed_data.timevector.
489        """
490        self.recording = recording
491        self.data_path = data_path
492        self._meta_path = meta_path  # self.meta_path is a property
493        self.load_path = load_path
494
495        self.metadata = metadata
496        if annotations:
497            self.update(annotations)
498        # else:
499        #     self.annotations = {}
500
501        if parsed_data:
502            self._modality_data = parsed_data
503        else:
504            self._modality_data = None
505
506        if (self._modality_data is None or
507                self._modality_data.timevector is None):
508            self._time_offset = time_offset
509        else:
510            self._time_offset = self._modality_data.timevector[0]
511
512        # This is a property which when set to True will also set
513        # parent.excluded to True.
514        self.excluded = False
515        self._time_precision = None
516
517    @property
518    def annotations(self) -> dict[AnnotationType, PointAnnotations]:
519        """
520        Property which is the annotations dictionary of this Modality.
521
522        This is currently the Modality itself.
523
524        Returns
525        -------
526        dict[AnnotationType, PointAnnotations]
527            The dictionary.
528        """
529        return self

531    def _get_data(self) -> ModalityData:
532        # TODO: Provide a way to force the data to be derived or loaded.
533        # this would be used when parent modality has updated in some way
534        if self.data_path:
535            return self._read_data()
536
537        if self.load_path:
538            return self._load_data()
539
540        if self.metadata.parent_name:
541            return self._derive_data()
542        else:
543            raise MissingDataError(
544                "Asked to get data but have no path and no parent Modality.\n"
545                + "Don't know how to solve this.")

557        This method is intended to rely on self.parent to provide the data

591    def add_annotations(self, annotations: PointAnnotations) -> None:

593        Add the PointAnnotations object to this Modality.

619        """

635    @property
636    def modality_data(self) -> ModalityData:
637        """
638        The data of this Modality as a NumPy array.

641        and for DerivedModality it is the result of running the 
642        modality's algorithm on the original data.
643
644        The dimensions of the array are in the 
645        order of [time, others]
646
647        If this modality is not preloaded, accessing this property will
648        cause data to be loaded on the fly _and_ saved in memory. To 
649        release the memory, assign None to this Modality's data.
650        """
651        if self._modality_data is None:
652            _datastructures_logger.debug(
653                "In Modality modality_data getter. "
654                "self._modality_data was None.")
655            self._set_data(self._get_data())
656        return self._modality_data

699    def _data_setter(self, data: np.ndarray) -> None:
